# bib to yaml

pandoc rvw-biblio/RVW-Publikationen.bib -s -f biblatex -t markdown-smart --template rvw-biblio/RVW-Publikationen-yaml-template.txt -o rvw-biblio/RVW-Publikationen.yaml


# yaml to html for hugo

pandoc --from=markdown rvw-biblio/RVW-web.md -C --biblio=rvw-biblio/RVW-Publikationen.yaml --csl=rvw-biblio/RVW-web.csl --lua-filter=rvw-biblio/RVW-web.lua --to=html5 -o rvw-biblio/RVW-web.htm
